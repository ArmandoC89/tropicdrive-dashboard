This replaces your current script. It writes to 3 tabs: ads_spend (totals, same as before) + ads_campaigns (per-campaign) + ads_keywords (top keywords by spend). 
// ══════════════════════════════════════════
// TropicDrive — Google Ads → Sheets (Full)
// Tabs: ads_spend | ads_campaigns | ads_keywords
// ══════════════════════════════════════════

var SPREADSHEET_URL = "YOUR_SPREADSHEET_URL_HERE";

function main() {
  var ss        = SpreadsheetApp.openByUrl(SPREADSHEET_URL);
  var shSpend   = ss.getSheetByName("ads_spend")     || ss.insertSheet("ads_spend");
  var shCamp    = ss.getSheetByName("ads_campaigns") || ss.insertSheet("ads_campaigns");
  var shKw      = ss.getSheetByName("ads_keywords")  || ss.insertSheet("ads_keywords");

  var yesterday = new Date();
  yesterday.setDate(yesterday.getDate() - 1);
  var d = Utilities.formatDate(yesterday, "UTC", "yyyy-MM-dd");

  // ── Ensure headers ──────────────────────
  if (shSpend.getLastRow() === 0)
    shSpend.appendRow(["date","spend_usd","clicks","impressions","conversions","currency"]);
  if (shCamp.getLastRow() === 0)
    shCamp.appendRow(["date","campaign","spend_usd","clicks","impressions","ctr","avg_cpc","conversions","status"]);
  if (shKw.getLastRow() === 0)
    shKw.appendRow(["date","keyword","match_type","campaign","spend_usd","clicks","impressions","avg_cpc","conversions","quality_score"]);

  // ── Exchange rate: EUR → USD ─────────────
  var ratesResp = UrlFetchApp.fetch("https://api.exchangerate-api.com/v4/latest/USD");
  var eurToUsd  = 1 / JSON.parse(ratesResp.getContentText()).rates["EUR"];

  // ── 1. TOTALS (ads_spend) ────────────────
  var totalReport = AdsApp.report(
    "SELECT metrics.cost_micros, metrics.clicks, metrics.impressions, metrics.conversions " +
    "FROM campaign WHERE segments.date = '" + d + "'"
  );
  var totCost=0, totClicks=0, totImps=0, totConv=0;
  var rows = totalReport.rows();
  while (rows.hasNext()) {
    var r = rows.next();
    totCost   += parseInt(r["metrics.cost_micros"]) / 1e6;
    totClicks += parseInt(r["metrics.clicks"]);
    totImps   += parseInt(r["metrics.impressions"]);
    totConv   += parseFloat(r["metrics.conversions"]);
  }
  shSpend.appendRow([d, (totCost*eurToUsd).toFixed(2), totClicks, totImps, totConv.toFixed(2), "USD"]);

  // ── 2. CAMPAIGNS (ads_campaigns) ─────────
  var campReport = AdsApp.report(
    "SELECT campaign.name, campaign.status, " +
    "metrics.cost_micros, metrics.clicks, metrics.impressions, " +
    "metrics.ctr, metrics.average_cpc, metrics.conversions " +
    "FROM campaign WHERE segments.date = '" + d + "'"
  );
  var campRows = campReport.rows();
  while (campRows.hasNext()) {
    var cr   = campRows.next();
    var cost = parseInt(cr["metrics.cost_micros"]) / 1e6;
    var cpc  = parseInt(cr["metrics.average_cpc"]) / 1e6;
    shCamp.appendRow([
      d,
      cr["campaign.name"],
      (cost * eurToUsd).toFixed(2),
      parseInt(cr["metrics.clicks"]),
      parseInt(cr["metrics.impressions"]),
      parseFloat(cr["metrics.ctr"]).toFixed(4),
      (cpc  * eurToUsd).toFixed(2),
      parseFloat(cr["metrics.conversions"]).toFixed(2),
      cr["campaign.status"]
    ]);
  }

  // ── 3. KEYWORDS (ads_keywords, top 50 by spend) ──
  var kwReport = AdsApp.report(
    "SELECT ad_group_criterion.keyword.text, " +
    "ad_group_criterion.keyword.match_type, " +
    "campaign.name, " +
    "metrics.cost_micros, metrics.clicks, metrics.impressions, " +
    "metrics.average_cpc, metrics.conversions, " +
    "ad_group_criterion.quality_info.quality_score " +
    "FROM keyword_view " +
    "WHERE segments.date = '" + d + "' " +
    "AND metrics.cost_micros > 0 " +
    "ORDER BY metrics.cost_micros DESC " +
    "LIMIT 50"
  );
  var kwRows = kwReport.rows();
  while (kwRows.hasNext()) {
    var kw  = kwRows.next();
    var kc  = parseInt(kw["metrics.cost_micros"]) / 1e6;
    var kcp = parseInt(kw["metrics.average_cpc"]) / 1e6;
    shKw.appendRow([
      d,
      kw["ad_group_criterion.keyword.text"],
      kw["ad_group_criterion.keyword.match_type"],
      kw["campaign.name"],
      (kc  * eurToUsd).toFixed(2),
      parseInt(kw["metrics.clicks"]),
      parseInt(kw["metrics.impressions"]),
      (kcp * eurToUsd).toFixed(2),
      parseFloat(kw["metrics.conversions"]).toFixed(2),
      kw["ad_group_criterion.quality_info.quality_score"] || "—"
    ]);
  }

  Logger.log("Done: " + d + " | Total spend USD: " + (totCost*eurToUsd).toFixed(2));
}

② GA4 Apps Script — Traffic & Audience
PASTE AT script.google.com — SET DAILY TRIGGER AT 07:30
Creates & fills the ga4_daily tab with: sessions, users, new users, bounce rate, avg duration, top countries, device split, traffic sources. 
// ══════════════════════════════════════════
// TropicDrive — GA4 → Sheets
// Requires: Google Analytics Data API enabled
//           in your Apps Script project
// ══════════════════════════════════════════

var GA4_PROPERTY  = "YOUR_GA4_PROPERTY_ID"; // numbers only, e.g. 123456789
var SHEET_URL     = "YOUR_SPREADSHEET_URL";
var TAB_NAME      = "ga4_daily";

function runGA4Daily() {
  var ss    = SpreadsheetApp.openByUrl(SHEET_URL);
  var sheet = ss.getSheetByName(TAB_NAME) || ss.insertSheet(TAB_NAME);

  if (sheet.getLastRow() === 0)
    sheet.appendRow(["date","sessions","users","new_users","bounce_rate",
                     "avg_duration","countries","devices","sources"]);

  var yesterday = new Date();
  yesterday.setDate(yesterday.getDate() - 1);
  var d = Utilities.formatDate(yesterday, "UTC", "yyyy-MM-dd");

  var prop = "properties/" + GA4_PROPERTY;

  // Core metrics
  var core = AnalyticsData.Properties.runReport(prop, {
    dateRanges: [{startDate: d, endDate: d}],
    metrics: [
      {name:"sessions"},{name:"activeUsers"},
      {name:"newUsers"},{name:"bounceRate"},
      {name:"averageSessionDuration"}
    ]
  });
  var m = core.rows ? core.rows[0].metricValues : null;
  if (!m) { Logger.log("No GA4 data for " + d); return; }

  // Countries (top 5)
  var geo = AnalyticsData.Properties.runReport(prop, {
    dateRanges: [{startDate: d, endDate: d}],
    metrics: [{name:"sessions"}],
    dimensions: [{name:"country"}],
    orderBys: [{metric:{metricName:"sessions"},desc:true}],
    limit: 5
  });

  // Devices
  var dev = AnalyticsData.Properties.runReport(prop, {
    dateRanges: [{startDate: d, endDate: d}],
    metrics: [{name:"sessions"}],
    dimensions: [{name:"deviceCategory"}]
  });

  // Traffic sources
  var src = AnalyticsData.Properties.runReport(prop, {
    dateRanges: [{startDate: d, endDate: d}],
    metrics: [{name:"sessions"}],
    dimensions: [{name:"sessionDefaultChannelGroup"}],
    orderBys: [{metric:{metricName:"sessions"},desc:true}],
    limit: 6
  });

  var geoStr = (geo.rows||[]).map(function(r){
    return r.dimensionValues[0].value+":"+r.metricValues[0].value;
  }).join("|");

  var devStr = (dev.rows||[]).map(function(r){
    return r.dimensionValues[0].value+":"+r.metricValues[0].value;
  }).join("|");

  var srcStr = (src.rows||[]).map(function(r){
    return r.dimensionValues[0].value+":"+r.metricValues[0].value;
  }).join("|");

  sheet.appendRow([
    d,
    parseInt(m[0].value),
    parseInt(m[1].value),
    parseInt(m[2].value),
    parseFloat(m[3].value).toFixed(3),
    parseFloat(m[4].value).toFixed(1),
    geoStr, devStr, srcStr
  ]);
  Logger.log("GA4 done: " + d);
}

③ Monthly Summary — Google Sheets Formulas
CREATE A NEW TAB CALLED monthly_summary AND PASTE THESE FORMULAS IN ROW 2
This tab auto-calculates monthly totals from your existing stripe_daily and ads_spend data. No script needed — pure formulas. 
// ── Tab: monthly_summary ──────────────────────────────
// Headers (paste in row 1):
// month | gross_revenue | ad_spend | profit | bookings | avg_booking_value | roas | sessions

// In row 2, col A — generates last 18 months automatically:
=ARRAYFORMULA(TEXT(EOMONTH(TODAY(),-ROW(INDIRECT("A1:A18"))+1)+1,"YYYY-MM"))

// In row 2, col B (gross_revenue):
=IFERROR(SUMPRODUCT((TEXT(DATEVALUE(stripe_daily!A2:A),"YYYY-MM")=A2)*stripe_daily!B2:B),0)

// In row 2, col C (ad_spend):
=IFERROR(SUMPRODUCT((TEXT(DATEVALUE(ads_spend!A2:A),"YYYY-MM")=A2)*ads_spend!B2:B),0)

// In row 2, col D (profit):
=B2-C2

// In row 2, col E (bookings):
=IFERROR(SUMPRODUCT((TEXT(DATEVALUE(stripe_daily!A2:A),"YYYY-MM")=A2)*stripe_daily!D2:D),0)

// In row 2, col F (avg booking value):
=IFERROR(B2/E2,0)

// In row 2, col G (roas):
=IFERROR(IF(C2>0,B2/C2,0),0)

// In row 2, col H (sessions — once GA4 is connected):
=IFERROR(SUMPRODUCT((TEXT(DATEVALUE(ga4_daily!A2:A),"YYYY-MM")=A2)*ga4_daily!B2:B),0)

// ── IMPORTANT ─────────────────────────────────────────
// Drag rows 2 down through row 19 (covers 18 months).
// Or use Dutch syntax (semicolons) if your Sheet is set to Dutch locale.
// Dutch example for col B:
// =ALS.FOUT(SOMPRODUCT((TEKST(DATUMWAARDE(stripe_daily!A2:A);"JJJJ-MM")=A2)*stripe_daily!B2:B);0)
